  2. Write code to explain prototype chaining

    3. Answer:

              let animal = {
            type: 'animal',
            makeSound: function() {
              console.log("Some generic animal sound");
            }
          };
          
          let dog = {
            type: 'dog',
            makeSound: function() {
              console.log("Woof!");
            }
          };

          Object.setPrototypeOf(dog, animal);

          let test = Object.create(dog);
console.log(test.type); // Output: dog
test.makeSound();
